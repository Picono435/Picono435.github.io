196        public boolean generateJobsFromConfig() {
197                jobs.clear();
198                ConfigurationSection jobsc = FileCreator.getJobsConfig().getConfigurationSection("jobs");
199                for(String jobid : jobsc.getKeys(false)) {
200                        ConfigurationSection jobc = jobsc.getConfigurationSection(jobid);
201                        String displayname = jobc.getString("displayname");
202                        String tag = jobc.getString("tag");
203                        String typeString = jobc.getString("type");
204                        Type type = Type.getType(typeString.toUpperCase());
205                        double method = jobc.getDouble(PicoJobsAPI.getJobsManager().getConfigMethod(type));
206                        double salary = jobc.getDouble("salary");
207                        boolean requiresPermission = jobc.getBoolean("require-permission");
208                        double salaryFrequency = jobc.getDouble("salary-frequency");
209                        double methodFrequency = jobc.getDouble("method-frequency");
210                        String economy = jobc.getString("economy");
211                        if(economy != null) {
212                                economy = economy.toUpperCase();
213                        }
214                        String workMessage = jobc.getString("work-message");
215                        ConfigurationSection guic = jobc.getConfigurationSection("gui");
216                        int slot = guic.getInt("slot");
217                        String item = guic.getString("item");
218                        int itemData = guic.getInt("item-data");
219                        boolean enchanted = guic.getBoolean("enchanted");
220                        
221                        // CALCULATING OPTIONALS
222                        
223                        boolean useWhitelist = jobc.getBoolean("use-whitelist");
224                        List<String> whitelist = jobc.getStringList(PicoJobsAPI.getJobsManager().getConfigWhitelistString(type));
225                        
226                        Job job = new Job(jobid, displayname, tag, type, method, salary, requiresPermission, salaryFrequency, methodFrequency, economy, workMessage, slot, item, itemData, enchanted, useWhitelist, whitelist);
227                        jobs.put(jobid, job);
228                                                
229                        metrics.addCustomChart(new Metrics.DrilldownPie("jobs", () -> {
230                        Map<String, Map<String, Integer>> map = new HashMap<>();
231                        Map<String, Integer> entry = new HashMap<>();
232                        entry.put(jobid, 1);
233                        map.put(type.name(), entry);
234                        return map;
235                    }));
236
237                        metrics.addCustomChart(new Metrics.DrilldownPie("active_economy", () -> {
238                        Map<String, Map<String, Integer>> map = new HashMap<>();
239                        Map<String, Integer> entry = new HashMap<>();
240                        String eco = job.getEconomy();
241                        if(eco.equalsIgnoreCase("VAULT")) {
242                                if(VaultHook.isEnabled() && VaultHook.hasEconomyPlugin()) {
243                                        entry.put(VaultHook.getEconomy().getName(), 1);
244                                } else {
245                                        entry.put("Others", 1);
246                                }
247                                map.put("VAULT", entry);
248                        } else {
249                                entry.put(eco, 1);
250                                map.put(eco, entry);
251                        }
252                        return map;
253                    }));
254                }
255                return true;
256        }

258        private boolean checkLegacy() {
259                try {
260                        String serverVersionString = Bukkit.getBukkitVersion();
261                        int spaceIndex = serverVersionString.indexOf("-");
262                        serverVersionString = serverVersionString.substring(0, spaceIndex);
263                        DefaultArtifactVersion legacyVersion = new DefaultArtifactVersion("1.13.2");
264                        DefaultArtifactVersion serverVersion = new DefaultArtifactVersion(serverVersionString);
265                        if(serverVersion.compareTo(legacyVersion) <= 0) {
266                                legacy = true;
267                        }
268                        return legacy;
269                } catch (Exception e) {
270                        return legacy;
271                }
272        }

274        private void checkVersion() {
275                String version = "1.0";
276                try {
277            URL url = new URL("https://servermods.forgesvc.net/servermods/files?projectIds=385252");
278            HttpURLConnection con = (HttpURLConnection) url.openConnection();
279            con.setRequestMethod("GET");
280            con.setConnectTimeout(5000);
281            con.setReadTimeout(5000);
282            con.setInstanceFollowRedirects(false);
283
284            BufferedReader in = new BufferedReader(new InputStreamReader(con.getInputStream()));
285            String inputLine;
286            StringBuffer content = new StringBuffer();
287            while ((inputLine = in.readLine()) != null) {
288                content.append(inputLine);
289            }
290            in.close();
291
292            JSONParser parser = new JSONParser();
293            JSONArray jsonArray = (JSONArray) parser.parse(content.toString());
294            JSONObject json = (JSONObject) jsonArray.get(jsonArray.size() - 1);
295            version = (String)json.get("name");
296            version = version.replaceFirst("PicoJobs ", "");
297
298                        DefaultArtifactVersion pluginVersion = new DefaultArtifactVersion(getDescription().getVersion());
299                        DefaultArtifactVersion lastestVersion = new DefaultArtifactVersion(version);
300                        lastestPluginVersion = version;
301                        downloadUrl = (String)json.get("downloadUrl");
302                        if(lastestVersion.compareTo(pluginVersion) > 0) {
303                                new BukkitRunnable() {
304                                        public void run() {
305                                                sendConsoleMessage(Level.WARNING, "Version: " + lastestVersion.toString() + " is out! You are still running version: " + pluginVersion.toString());
306                                                if(getConfig().getBoolean("auto-update")) {
307                                                        if(updatePlugin(Bukkit.getConsoleSender(), "[PicoJobs] Plugin was updated to version "+ lastestVersion.toString() + " sucefully. Please restart the server to finish the update.")) {
308                                                                sendConsoleMessage(Level.INFO, "Updating the plugin to the lastest version...");
309                                                        } else {
310                                                                sendConsoleMessage(Level.WARNING, "An error occuried while updating the plugin.");
311                                                        }
312                                                }
313                                                return;
314                                        }
315                                }.runTaskLater(this, 40L);
316                                oldVersion = true;
317                        } else {
318                                new BukkitRunnable() {
319                                        public void run() {
320                                                sendConsoleMessage(Level.INFO, "You are using the lastest version of the plugin.");
321                                                return;
322                                        }
323                                }.runTaskLater(this, 40L);
324                        }
325                } catch (Exception e) {
326                        sendConsoleMessage(Level.WARNING, "Could not get the lastest version.");
327                        return;
328                }
329        }

331        public boolean updatePlugin(CommandSender p, String message) {
332                try {
333                        URL url = new URL(PicoJobsPlugin.getInstance().getLastestDownloadUrl());
334                        
335                        Method getFileMethod = JavaPlugin.class.getDeclaredMethod("getFile");
336                        getFileMethod.setAccessible(true);
337                        File oldFile = (File) getFileMethod.invoke(PicoJobsPlugin.getInstance());
338
339                        File fileOutput = new File(Bukkit.getUpdateFolderFile().getPath() + File.separatorChar + oldFile.getName());
340                        if(!fileOutput.exists()) {
341                                fileOutput.mkdirs();
342                        }
343                        
344                        downloadFile(url, fileOutput, p, message);
345                        return true;
346                } catch(Exception ex) {
347                        ex.printStackTrace();
348                        return false;
349                }
350        }
351        
352        private void downloadFile(URL url, File fileOutput, CommandSender p, String message) {
353                new BukkitRunnable() {
354                        public void run() {
355                                try {
356                                        if (fileOutput.exists()) {
357                                                fileOutput.delete();
358                                    }
359                                        fileOutput.createNewFile();
360                                    OutputStream out = new BufferedOutputStream(new FileOutputStream(fileOutput.getPath()));
361                                    URLConnection conn = url.openConnection();
362                                    String encoded = Base64.getEncoder().encodeToString(("username"+":"+"password").getBytes(StandardCharsets.UTF_8));  //Java 8
363                                    conn.setRequestProperty("Authorization", "Basic "+ encoded);
364                                    InputStream in = conn.getInputStream();
365                                    byte[] buffer = new byte[1024];
366
367                                    int numRead;
368                                    while ((numRead = in.read(buffer)) != -1) {
369                                        out.write(buffer, 0, numRead);
370                                    }
371                                    if (in != null) {
372                                        in.close();
373                                    }
374                                    if (out != null) {
375                                        out.close();
376                                    }
377                                    p.sendMessage(message);
378                                    oldVersion = false;
379                                } catch(Exception ex) {
380                                        ex.printStackTrace();
381                                }
382                        }
383                }.runTaskAsynchronously(PicoJobsPlugin.getInstance());
384        }
