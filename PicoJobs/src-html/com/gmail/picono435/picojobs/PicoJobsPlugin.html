176        private static boolean generateJobsFromConfig() {
177                ConfigurationSection jobsc = FileCreator.getJobsConfig().getConfigurationSection("jobs");
178                for(String jobname : jobsc.getKeys(false)) {
179                        ConfigurationSection jobc = jobsc.getConfigurationSection(jobname);
180                        String displayname = jobc.getString("displayname");
181                        String tag = jobc.getString("tag");
182                        String typeString = jobc.getString("type");
183                        Type type = Type.getType(typeString.toUpperCase());
184                        double method = getJobMethodFromConfig(jobname, type);
185                        double salary = jobc.getDouble("salary");
186                        boolean requiresPermission = jobc.getBoolean("require-permission");
187                        double salaryFrequency = jobc.getDouble("salary-frequency");
188                        double methodFrequency = jobc.getDouble("method-frequency");
189                        ConfigurationSection guic = jobc.getConfigurationSection("gui");
190                        int slot = guic.getInt("slot");
191                        String item = guic.getString("item");
192                        int itemData = guic.getInt("item-data");
193                        boolean enchanted = guic.getBoolean("enchanted");
194                        
195                        // CALCULATING OPTIONALS
196                        
197                        String killJob = "";
198                        if(type == Type.KILL && jobc.getString("kill-job") != null && !jobc.getString("kill-job").equalsIgnoreCase("all")) {
199                                killJob = jobc.getString("kill-job");
200                        }
201                        
202                        boolean useWhitelist = jobc.getBoolean("use-whitelist");
203                        List<String> blockWhitelist = null;
204                        if(type == Type.BREAK || type == Type.PLACE) {
205                                blockWhitelist = jobc.getStringList("block-whitelist");
206                        }
207                        if(type == Type.CRAFT || type == Type.SMELT || type == Type.ENCHANTING || type == Type.REPAIR || type == Type.EAT) {
208                                blockWhitelist = jobc.getStringList("item-whitelist");
209                        }
210                        
211                        Job job = new Job(jobname, displayname, tag, type, method, salary, requiresPermission, salaryFrequency, methodFrequency, slot, item, itemData, enchanted, killJob, useWhitelist, blockWhitelist);
212                        jobs.put(jobname, job);
213                }
214                return true;
215        }
216        
217        private static double getJobMethodFromConfig(String jobname, Type type) {
218                ConfigurationSection cat =  FileCreator.getJobsConfig().getConfigurationSection("jobs").getConfigurationSection(jobname);
219                if(type == Type.BREAK || type == Type.PLACE) {
220                        return cat.getDouble("blocks");
221                }
222                if(type == Type.KILL) {
223                        return cat.getDouble("kills");
224                }
225                if(type == Type.FISHING) {
226                        return cat.getDouble("fish");
227                }
228                if(type == Type.CRAFT || type == Type.SMELT || type == Type.ENCHANTING || type == Type.REPAIR) {
229                        return cat.getDouble("items");
230                }
231                if(type == Type.EAT) {
232                        return cat.getDouble("food");
233                }
234                if(type == Type.MILK) {
235                        return cat.getDouble("buckets");
236                }
237                return 0.0;
238        }

247        private static boolean checkLegacy() {
248                try {
249                        String serverVersionString = Bukkit.getBukkitVersion();
250                        int spaceIndex = serverVersionString.indexOf("-");
251                        serverVersionString = serverVersionString.substring(0, spaceIndex);
252                        DefaultArtifactVersion legacyVersion = new DefaultArtifactVersion("1.13.2");
253                        DefaultArtifactVersion serverVersion = new DefaultArtifactVersion(serverVersionString);
254                        if(serverVersion.compareTo(legacyVersion) <= 0) {
255                                legacy = true;
256                        }
257                        return legacy;
258                } catch (Exception e) {
259                        return legacy;
260                }
261        }

263        private void checkVersion() {
264                String version = "1.0";
265                try {
266            URL url = new URL("https://api.github.com/repos/Picono435/PicoJobs/releases");
267            HttpURLConnection con = (HttpURLConnection) url.openConnection();
268            con.setRequestMethod("GET");
269            con.setConnectTimeout(5000);
270            con.setReadTimeout(5000);
271            con.setInstanceFollowRedirects(false);
272
273            BufferedReader in = new BufferedReader(new InputStreamReader(con.getInputStream()));
274            String inputLine;
275            StringBuffer content = new StringBuffer();
276            while ((inputLine = in.readLine()) != null) {
277                content.append(inputLine);
278            }
279            in.close();
280
281            JSONParser parser = new JSONParser();
282            JSONArray jsonArray = (JSONArray) parser.parse(content.toString());
283            JSONObject json = (JSONObject) jsonArray.get(0);
284            version = (String)json.get("tag_name");
285                        
286                } catch(Exception ex) {
287                        sendConsoleMessage(ChatColor.DARK_RED + "[PicoJobs] Could not get the lastest version.");
288                        return;
289                }
290                try {
291                        DefaultArtifactVersion pluginVesion = new DefaultArtifactVersion(getPlugin().getDescription().getVersion());
292                        DefaultArtifactVersion lastestVersion = new DefaultArtifactVersion(version);
293                        lastestPluginVersion = version;
294                        if(lastestVersion.compareTo(pluginVesion) > 0) {
295                                new BukkitRunnable() {
296                                        public void run() {
297                                                sendConsoleMessage(ChatColor.DARK_RED + "[PicoJobs] You are using a old version of the plugin. Please download the new version in our pages.");
298                                                oldVersion = true;
299                                                return;
300                                        }
301                                }.runTaskLater(this, 5L);
302                        } else {
303                                new BukkitRunnable() {
304                                        public void run() {
305                                                sendConsoleMessage(ChatColor.GREEN + "[PicoJobs] You are using the lastest version of the plugin.");
306                                                return;
307                                        }
308                                }.runTaskLater(this, 5L);
309                        }
310                } catch (Exception e) {
311                        sendConsoleMessage(ChatColor.DARK_RED + "[PicoJobs] Could not get the lastest version.");
312                        return;
313                }
314        }
