001package com.gmail.picono435.picojobs.api;
002
003import org.bukkit.Material;
004import org.bukkit.enchantments.Enchantment;
005import org.bukkit.inventory.ItemStack;
006
007import com.gmail.picono435.picojobs.PicoJobsPlugin;
008import com.gmail.picono435.picojobs.utils.ItemBuilder;
009
010import net.md_5.bungee.api.ChatColor;
011
012/**
013 * Represents a job.
014 * 
015 * @author Picono435
016 *
017 */
018public class Job {
019        
020        private String name;
021        private String displayname;
022        private String tag;
023        private Type type;
024        private double method;
025        private double salary;
026        private boolean requiresPermission;
027        
028        // GUI SETTINGS
029        private int slot;
030        private Material item;
031        private int itemData;
032        private boolean enchanted;
033        
034        // OPTIONAL
035        private String killJob;
036        
037        public Job(String name, String displayname, String tag, Type type, double method, double salary, boolean requiresPermission, int slot, String item, int itemData, boolean enchanted, String killJob) {
038                this.name = name;
039                this.displayname = displayname;
040                this.tag = tag;
041                this.type = type;
042                this.method = method;
043                this.salary = salary;
044                this.requiresPermission = requiresPermission;
045                
046                this.slot = slot;
047                Material m;
048                if(PicoJobsPlugin.isLegacy()) {
049                        m = Material.getMaterial(item.toUpperCase());
050                } else {
051                        m = Material.getMaterial(item.toUpperCase());
052                }
053                this.item = m;
054                this.itemData = itemData;
055                this.enchanted = enchanted;
056                
057                this.killJob = killJob;
058        }
059        
060        /**
061         * Gets the name of the job
062         * 
063         * @return the name of the job
064         * @author Picono435
065         */
066        public String getName() {
067                return name;
068        }
069        
070        /**
071         * Gets the display name of the job
072         * 
073         * @return the display name of the job
074         * @author Picono435
075         */
076        public String getDisplayName() {
077                return ChatColor.translateAlternateColorCodes('&', displayname);
078        }
079        
080        /**
081         * Gets the tag of the job
082         * 
083         * @return the tag of the job
084         * @author Picono435
085         */
086        public String getTag() {
087                return ChatColor.translateAlternateColorCodes('&', tag);
088        }
089        
090        /**
091         * Gets the type of the job
092         * 
093         * @return the type of the job
094         * @author Picono435
095         */
096        public Type getType() {
097                return type;
098        }
099        
100        /**
101         * Gets the method of the job
102         * 
103         * @return the method of the job
104         * @author Picono435
105         */
106        public double getMethod() {
107                return method;
108        }
109        
110        /**
111         * Gets the salary of the job
112         * 
113         * @return the salary of the job
114         * @author Picono435
115         */
116        public double getSalary() {
117                return salary;
118        }
119        
120        /**
121         * Check if the job requires permission to enter or not
122         * 
123         * @return true if it requires permission, false if not
124         * @author Picono435
125         */
126        public boolean requiresPermission() {
127                return requiresPermission;
128        }
129        
130        /**
131         * Gets the item slot
132         * 
133         * @return the slot of the job item
134         * @author Picono435
135         */
136        public int getSlot() {
137                return slot - 1;
138        }
139        
140        /**
141         * Gets the item material
142         * 
143         * @return the material of the job item
144         * @author Picono435
145         */
146        public Material getMaterial() {
147                return item;
148        }
149        
150        /**
151         * Gets the item data slot
152         * 
153         * @deprecated
154         * @return the item data of the job item
155         * @author Picono435
156         */
157        @Deprecated
158        public int getItemData() {
159                return itemData;
160        }
161        
162        /**
163         * Checks if the job item is enchanted or not
164         * 
165         * @return true if the job item is enchanted, false if not
166         * @author Picono435
167         */
168        public boolean isEnchanted() {
169                return enchanted;
170        }
171        
172        
173        /**
174         * Gets the formatted item of the job
175         * 
176         * @return the formatted item
177         * @author Picono435
178         */
179        @SuppressWarnings("deprecation")
180        public ItemStack getFormattedItem() {
181                ItemBuilder builder;
182                if(PicoJobsPlugin.isLegacy()) {
183                        int itemData = getItemData() - 1;
184                        if(itemData == -1) {
185                                builder = new ItemBuilder(getMaterial());
186                        } else {
187                                builder = new ItemBuilder(getMaterial(), 1, (byte)itemData);
188                        }
189                } else {
190                        builder = new ItemBuilder(getMaterial());
191                }
192                builder.setName(getDisplayName());
193                if(isEnchanted()) builder.addUnsafeEnchantment(Enchantment.ARROW_DAMAGE, 1);
194                builder.removeAttributes();
195                return builder.toItemStack();
196        }
197        
198        public String getKillJob() {
199                return killJob;
200        }
201}
