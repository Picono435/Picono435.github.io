001package com.gmail.picono435.picojobs.api;
002
003import java.util.UUID;
004
005import org.bukkit.Bukkit;
006
007import com.gmail.picono435.picojobs.events.PlayerEnterJobEvent;
008import com.gmail.picono435.picojobs.events.PlayerFinishWorkEvent;
009import com.gmail.picono435.picojobs.events.PlayerLeaveJobEvent;
010import com.gmail.picono435.picojobs.events.PlayerStartWorkEvent;
011import com.gmail.picono435.picojobs.events.PlayerWithdrawEvent;
012
013/**
014 * Represents a player from PicoJobs
015 * 
016 * @author Picono435
017 *
018 */
019public class JobPlayer {
020        
021        private Job job;
022        private double method;
023        private double level;
024        private boolean isWorking;
025        private double salary;
026        private UUID uuid;
027        
028        public JobPlayer(Job job, double method, double level, double salary, boolean isWorking, UUID uuid) {
029                this.job = job;
030                this.method = method;
031                this.level = level;
032                this.salary = salary;
033                this.isWorking = isWorking;
034        }
035        
036        /**
037         * Gets the UUID of the player
038         * 
039         * @return the uuid
040         * @author Picono435
041         *
042         */
043        public UUID getUUID() {
044                return uuid;
045        }
046        
047        /**
048         * Check if a player has a job.
049         * 
050         * @return true if has job, false if doesn't have
051         * @author Picono435
052         *
053         */
054        public boolean hasJob() {
055                return job != null;
056        }
057        
058        /**
059         * Gets the job of a player.
060         * 
061         * @return null if player does not have job, the job if it does
062         * @author Picono435
063         *
064         */
065        public Job getJob() {
066                return job;
067        }
068        
069        /**
070         * Get the current method (broken blocks, fish caugh etc..) from a player
071         * 
072         * @return the method
073         * @author Picono435
074         *
075         */
076        public double getMethod() {
077                return method;
078        }
079        
080        /**
081         * Sets the current method (broken blocks, fish caugh etc..) of a player
082         * 
083         * @param method The method
084         * @author Picono435
085         *
086         */
087        public void setMethod(double method) {
088                this.method = method;
089        }
090        
091        /**
092         * Get the current method level (amount of works done) from a player
093         * 
094         * @return the method
095         * @author Picono435
096         *
097         */
098        public double getMethodLevel() {
099                return level;
100        }
101        
102        /**
103         * Sets the current method level (amount of works done) of a player
104         * 
105         * @param level the level
106         * @author Picono435
107         *
108         */
109        public void setMethodLevel(double level) {
110                this.level = level;
111        }
112        
113        /**
114         * Sets the job of a player
115         * 
116         * @param job The Job
117         * @author Picono435
118         *
119         */
120        public void setJob(Job job) {
121                if(job == null) {
122                        PlayerLeaveJobEvent event = new PlayerLeaveJobEvent(this, Bukkit.getPlayer(uuid), job);
123                        Bukkit.getPluginManager().callEvent(event);
124                }
125                PlayerEnterJobEvent event = new PlayerEnterJobEvent(this, Bukkit.getPlayer(uuid), job);
126                Bukkit.getPluginManager().callEvent(event);
127                if(event.isCancelled()) {
128                        return;
129                }
130                this.job = job;
131        }
132        
133        /**
134         * Check if a player has accepted a work or not
135         * 
136         * @return true if it is working, false if not
137         * @author Picono435
138         *
139         */
140        public boolean isWorking() {
141                return isWorking;
142        }
143        
144        /**
145         * Set if the player is working or not
146         * 
147         * @param isWorking if it's working or not
148         * @author Picono435
149         *
150         */
151        public void setWorking(boolean isWorking) {
152                if(isWorking) {
153                        PlayerStartWorkEvent event = new PlayerStartWorkEvent(this, Bukkit.getPlayer(uuid), getJob());
154                        Bukkit.getPluginManager().callEvent(event);
155                }
156                this.isWorking = isWorking;
157        }
158        
159        /**
160         * Gets the current salary of a player
161         * 
162         * @return the salary
163         * @author Picono435
164         *
165         */
166        public double getSalary() {
167                return salary;
168        }
169        
170        /**
171         * Sets the current salary of a player
172         * 
173         * @param salary the salary to set
174         * @author Picono435
175         *
176         */
177        public void setSalary(double salary) {
178                this.salary = salary;
179        }
180        
181        /**
182         * Adds some salary from the current salary of a player
183         * 
184         * @param salary the salary to add
185         * @author Picono435
186         *
187         */
188        public void addSalary(double salary) {
189                setSalary(getSalary() + salary);
190        }
191        
192        /**
193         * Removes some salary from the current salary of a player
194         * 
195         * @param salary the salary to remove
196         * @author Picono435
197         *
198         */
199        public void removeSalary(double salary) {
200                PlayerWithdrawEvent event = new PlayerWithdrawEvent(this, Bukkit.getPlayer(uuid), salary);
201                Bukkit.getPluginManager().callEvent(event);
202                setSalary(getSalary() - salary);
203        }
204        
205        /**
206         * Simulates an event, for example it will simulate a caught fish or a break block from the plugin.
207         * 
208         * @return true if it will level up, false if not
209         * @author Picono435
210         *
211         */
212        public boolean simulateEvent(Type type) {
213                if(!isWorking()) return false;
214                double level = getMethodLevel();
215                double method = getMethod();
216                setMethod(method + 1);
217                                
218                int reqmethod = (int) (job.getMethod() * level * getJob().getMethodFrequency());
219                
220                if(getMethod() >= reqmethod) {
221                        PlayerFinishWorkEvent event = new PlayerFinishWorkEvent(this, Bukkit.getPlayer(uuid), getJob());
222                        Bukkit.getPluginManager().callEvent(event);
223                        if(event.isCancelled()) {
224                                return false;
225                        }
226                        double salary = job.getSalary() * level * getJob().getSalaryFrequency();
227                        setMethodLevel(level + 1);
228                        setMethod(0);
229                        setWorking(false);
230                        setSalary(getSalary() + salary);
231                        return true;
232                }
233                return false;
234        }
235        
236        /**
237         * Resets completly the player, including the job, if its working the method and etc... but keeps half of the salary
238         * 
239         * @author Picono435
240         *
241         */
242        public void removePlayerStats() {
243                setWorking(false);
244                setSalary(getSalary() / 2);
245                setMethod(0);
246                setMethodLevel(1);
247                setJob(null);
248        }
249}
