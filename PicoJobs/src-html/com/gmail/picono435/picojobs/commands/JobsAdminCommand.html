028public class JobsAdminCommand implements CommandExecutor, TabCompleter {
029
030        @Override
031        public boolean onCommand(CommandSender sender, Command cmd, String label, String[] args) {
032                if(!cmd.getName().equals("jobsadmin")) return false;
033                CommandSender p = sender;
034                if(!p.hasPermission("picojobs.admin")) {
035                        p.sendMessage(LanguageManager.formatMessage("&7PicoJobs v" + PicoJobsPlugin.getInstance().getDescription().getVersion() + ". (&8&nhttps://discord.gg/wQj53Hy&r&7)"));
036                        return true;
037                }
038                if(args.length < 1) {
039                        p.sendMessage(LanguageManager.getFormat("admin-commands", null));
040                        return true;
041                }
042                String subcmd = args[0];
043                String helpString = LanguageManager.getSubCommandAlias("help");
044                String infoString = LanguageManager.getSubCommandAlias("info");
045                String reloadString = LanguageManager.getSubCommandAlias("reload");
046                String updateString = LanguageManager.getSubCommandAlias("update");
047                Player pl = null;
048                if(sender instanceof Player) {
049                        pl = (Player)sender;
050                }
051                
052                // HELP COMMAND
053                if(subcmd.equalsIgnoreCase("help") || subcmd.equalsIgnoreCase(helpString)) {
054                        p.sendMessage(LanguageManager.getFormat("admin-commands", pl));
055                        return true;
056                }
057                
058                // INFO COMMAND
059                if(subcmd.equalsIgnoreCase("info") || subcmd.equalsIgnoreCase(infoString)) {
060                        if(args.length < 2) {
061                                p.sendMessage(LanguageManager.getMessage("no-args", pl));
062                                return true;
063                        }
064                        String playername = args[1];
065                        Player player = Bukkit.getPlayer(playername);
066                        if(player == null) {
067                                p.sendMessage(LanguageManager.getMessage("player-not-found", pl));
068                                return true;
069                        }
070                        p.sendMessage(LanguageManager.getFormat("info-command", player));
071                        return true;
072                }
073                
074                // RELOAD COMMAND
075                if(subcmd.equalsIgnoreCase("reload") || subcmd.equalsIgnoreCase(reloadString)) {
076                        if(!FileCreator.reloadFiles()) {
077                                p.sendMessage(LanguageManager.getMessage("unknow-error", pl));
078                                return true;
079                        }
080                        p.sendMessage(LanguageManager.getMessage("reload-command", pl));
081                        return true;
082                }
083                
084                // RELOAD COMMAND
085                if(subcmd.equalsIgnoreCase("update") || subcmd.equalsIgnoreCase(updateString)) {
086                        if(!PicoJobsPlugin.isOldVersion()) {
087                                p.sendMessage(LanguageManager.getMessage("already-updated", pl));
088                                return true;
089                        }
090                        p.sendMessage(LanguageManager.getMessage("update-started", pl));
091                        if(!updatePlugin()) {
092                                p.sendMessage(LanguageManager.getMessage("unknow-error", pl));
093                                return true;
094                        }
095                        p.sendMessage(LanguageManager.getMessage("updated-sucefully", pl));
096                        return true;
097                }
098                return true;
099        }
100        
101        @Override
102        public List<String> onTabComplete(CommandSender sender, Command cmd, String alias, String[] args) {
103                Player p = (Player)sender;
104                if(cmd.getName().equalsIgnoreCase("jobsadmin")) {
105                        String helpString = LanguageManager.getSubCommandAlias("help");
106                        String infoString = LanguageManager.getSubCommandAlias("info");
107                        String reloadString = LanguageManager.getSubCommandAlias("reload");
108                        String updateString = LanguageManager.getSubCommandAlias("update");
109                        
110                        if(args.length == 1) {
111                                List<String> list = new ArrayList<String>();
112                                if(p.hasPermission("picojobs.admin")) {
113                                        list.add(helpString);
114                                        list.add(infoString);
115                                        list.add(reloadString);
116                                        list.add(updateString);
117                                }
118                                return list;
119                        }
120                }
121                return null;
122        }
123        
124        private boolean updatePlugin() {
125                try {
126                        PicoJobsPlugin.sendConsoleMessage(ChatColor.AQUA + "[PicoJobs] Updating the PicoJobs plugin to the version v" + PicoJobsPlugin.getLastestPluginVersion() + ". Please wait, the server may lag a little bit...");
127                        
128                        String downloadUrl = "https://github.com/Picono435/PicoJobs/releases/download/" + PicoJobsPlugin.getLastestPluginVersion() + "/PicoJobs-FREE-" + PicoJobsPlugin.getLastestPluginVersion() + ".jar";
129                        URL url = new URL(downloadUrl);
130                        File fileOutput = new File(PicoJobsPlugin.getInstance().getDataFolder().getParentFile().getPath() + File.separatorChar + "PicoJobs-FREE-" + PicoJobsPlugin.getLastestPluginVersion() + ".jar");
131                        
132                        downloadFile(url, fileOutput);
133                        
134            PicoJobsPlugin.sendConsoleMessage(ChatColor.AQUA + "[PicoJobs] Updated PicoJobs plugin to version v" + PicoJobsPlugin.getLastestPluginVersion() + " succefully.");
135                        return true;
136                } catch(Exception ex) {
137                        ex.printStackTrace();
138                        return false;
139                }
140        }
141        
142        public void downloadFile(URL url, File localFile) throws IOException {
143            if (localFile.exists()) {
144                localFile.delete();
145            }
146            localFile.createNewFile();
147            OutputStream out = new BufferedOutputStream(new FileOutputStream(localFile.getPath()));
148            URLConnection conn = url.openConnection();
149            String encoded = Base64.getEncoder().encodeToString(("username"+":"+"password").getBytes(StandardCharsets.UTF_8));  //Java 8
150            conn.setRequestProperty("Authorization", "Basic "+ encoded);
151            InputStream in = conn.getInputStream();
152            byte[] buffer = new byte[1024];
153
154            int numRead;
155            while ((numRead = in.read(buffer)) != -1) {
156                out.write(buffer, 0, numRead);
157            }
158            if (in != null) {
159                in.close();
160            }
161            if (out != null) {
162                out.close();
163            }
164        }
