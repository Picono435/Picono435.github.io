001package com.gmail.picono435.picojobs.events;
002
003import org.bukkit.entity.Player;
004import org.bukkit.event.Event;
005import org.bukkit.event.HandlerList;
006
007import com.gmail.picono435.picojobs.api.JobPlayer;
008
009public final class PlayerWithdrawEvent extends Event {
010    private static final HandlerList handlers = new HandlerList();
011    private JobPlayer jobPlayer;
012    private Player player;
013    private double salary;
014    
015    public PlayerWithdrawEvent(JobPlayer jobPlayer, Player player, double salary) {
016        this.jobPlayer = jobPlayer;
017        this.player = player;
018        this.salary = salary;
019    }
020
021    /**
022     * Gets the job player in this event
023     * 
024     * @return the job player
025     */
026    public JobPlayer getJobPlayer() {
027        return jobPlayer;
028    }
029    
030    /**
031     * Gets the player in this event
032     * 
033     * @return the player
034     */
035    public Player getPlayer() {
036        return player;
037    }
038   
039    /**
040     * Gets the salary took in this event
041     * 
042     * @return the salary took
043     */
044    public double getSalary() {
045        return salary;
046    }
047
048    public HandlerList getHandlers() {
049        return handlers;
050    }
051
052    public static HandlerList getHandlerList() {
053        return handlers;
054    }
055}
