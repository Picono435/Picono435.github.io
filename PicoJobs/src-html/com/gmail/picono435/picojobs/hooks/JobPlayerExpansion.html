001package com.gmail.picono435.picojobs.hooks;
002
003import java.text.NumberFormat;
004import java.util.Locale;
005
006import org.bukkit.entity.Player;
007
008import com.gmail.picono435.picojobs.PicoJobsPlugin;
009import com.gmail.picono435.picojobs.api.Job;
010import com.gmail.picono435.picojobs.api.JobPlayer;
011import com.gmail.picono435.picojobs.api.PicoJobsAPI;
012import com.gmail.picono435.picojobs.api.Type;
013import com.gmail.picono435.picojobs.managers.LanguageManager;
014
015import me.clip.placeholderapi.expansion.PlaceholderExpansion;
016
017/**
018 * This class will be registered through the register-method in the 
019 * plugins onEnable-method.
020 */
021public class JobPlayerExpansion extends PlaceholderExpansion {
022
023        NumberFormat df = NumberFormat.getNumberInstance(new Locale("pt", "BR"));
024    private PicoJobsPlugin plugin;
025
026    /**
027     * Since we register the expansion inside our own plugin, we
028     * can simply use this method here to get an instance of our
029     * plugin.
030     *
031     * @param plugin
032     *        The instance of our plugin.
033     */
034    public JobPlayerExpansion(PicoJobsPlugin plugin){
035        this.plugin = plugin;
036    }
037
038    /**
039     * Because this is an internal class,
040     * you must override this method to let PlaceholderAPI know to not unregister your expansion class when
041     * PlaceholderAPI is reloaded
042     *
043     * @return true to persist through reloads
044     */
045    @Override
046    public boolean persist(){
047        return true;
048    }
049
050    /**
051     * Because this is a internal class, this check is not needed
052     * and we can simply return {@code true}
053     *
054     * @return Always true since it's an internal class.
055     */
056    @Override
057    public boolean canRegister(){
058        return true;
059    }
060
061    /**
062     * The name of the person who created this expansion should go here.
063     * <br>For convienience do we return the author from the plugin.yml
064     * 
065     * @return The name of the author as a String.
066     */
067    @Override
068    public String getAuthor(){
069        return plugin.getDescription().getAuthors().toString();
070    }
071
072    /**
073     * The placeholder identifier should go here.
074     * <br>This is what tells PlaceholderAPI to call our onRequest 
075     * method to obtain a value if a placeholder starts with our 
076     * identifier.
077     * <br>The identifier has to be lowercase and can't contain _ or %
078     *
079     * @return The identifier in {@code %<identifier>_<value>%} as String.
080     */
081    @Override
082    public String getIdentifier(){
083        return "jobplayer";
084    }
085
086    /**
087     * This is the version of the expansion.
088     * <br>You don't have to use numbers, since it is set as a String.
089     *
090     * For convienience do we return the version from the plugin.yml
091     *
092     * @return The version as a String.
093     */
094    @Override
095    public String getVersion(){
096        return plugin.getDescription().getVersion();
097    }
098
099    /**
100     * This is the method called when a placeholder with our identifier 
101     * is found and needs a value.
102     * <br>We specify the value identifier in this method.
103     * <br>Since version 2.9.1 can you use OfflinePlayers in your requests.
104     *
105     * @param  p
106     *         A player.
107     * @param  identifier
108     *         A String containing the identifier/value.
109     *
110     * @return possibly-null String of the requested identifier.
111     */
112    @Override
113    public String onPlaceholderRequest(Player p, String identifier){
114
115        JobPlayer jp = PicoJobsAPI.getPlayersManager().getJobPlayer(p);
116        
117        if(identifier.equals("job")) {
118            if(!jp.hasJob()) {
119                return LanguageManager.getFormat("none-format", p);
120            }
121            return jp.getJob().getDisplayName();
122        }
123        
124        if(identifier.equals("work")) {
125                Job job = jp.getJob();
126                if(job == null) {
127                        return LanguageManager.getFormat("none-format", p);
128                }
129                Type type = job.getType();
130                String configString = type.name().toLowerCase() + "-work";
131                if(type == Type.KILL && !job.getKillJob().equals("")) {
132                        configString = "kill-specific-work";
133                }
134                double level = jp.getMethodLevel();
135                int reqmethod = (int) (job.getMethod() * level * PicoJobsAPI.getSettingsManager().getMethodFrequency());
136                double value = reqmethod - jp.getMethod();
137                String work = LanguageManager.getFormat("general-work", p);
138                work = work.replace("%a%", LanguageManager.getFormat(configString, p));
139                work = work.replace("%a%", df.format(value));
140                return work;
141        }
142        
143        if(identifier.equals("salary")) {
144            return df.format(jp.getSalary());
145        }
146        
147        if(identifier.equals("working")) {
148                return jp.isWorking() + "";
149        }
150
151        return "Incorrect Placeholder.";
152    }
153}
