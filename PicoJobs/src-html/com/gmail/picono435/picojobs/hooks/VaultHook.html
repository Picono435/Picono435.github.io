001package com.gmail.picono435.picojobs.hooks;
002
003import java.util.logging.Level;
004
005import org.bukkit.Bukkit;
006import org.bukkit.plugin.RegisteredServiceProvider;
007
008import com.gmail.picono435.picojobs.PicoJobsPlugin;
009import com.gmail.picono435.picojobs.api.PicoJobsAPI;
010import com.gmail.picono435.picojobs.hooks.economy.VaultImplementation;
011
012import net.milkbowl.vault.economy.Economy;
013
014public class VaultHook {
015        
016        private static Economy econ;
017        
018        private static boolean isEnabled = false;
019        private static boolean hasEconomy = true;
020        
021        public static void setupVault() {
022                if(Bukkit.getPluginManager().getPlugin("Vault") == null) {
023                        return;
024                }
025                isEnabled = true;
026                if(!setupEconomy()) {
027                        PicoJobsPlugin.getInstance().sendConsoleMessage(Level.WARNING, "A economy plugin from VAULT was not found. The VAULT economy type will not work.");
028                        hasEconomy = false;
029                }
030                
031                if(hasEconomy) {
032                        PicoJobsAPI.registerEconomy(new VaultImplementation());
033                }
034        }
035        
036        private static boolean setupEconomy() {
037        if (Bukkit.getServer().getPluginManager().getPlugin("Vault") == null) {
038            return false;
039        }
040        RegisteredServiceProvider<Economy> rsp = Bukkit.getServer().getServicesManager().getRegistration(Economy.class);
041        if (rsp == null) {
042            return false;
043        }
044        econ = rsp.getProvider();
045        return econ != null;
046    }
047        
048        public static Economy getEconomy() {
049                return econ;
050        }
051        
052        public static boolean hasEconomyPlugin() {
053                return hasEconomy;
054        }
055        
056        public static boolean isEnabled() {
057                return isEnabled;
058        }
059}
