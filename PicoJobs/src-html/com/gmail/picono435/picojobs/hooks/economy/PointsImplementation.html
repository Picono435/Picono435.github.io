001package com.gmail.picono435.picojobs.hooks.economy;
002
003import org.bukkit.Bukkit;
004import org.bukkit.entity.Player;
005
006import com.gmail.picono435.picojobs.api.EconomyImplementation;
007import com.gmail.picono435.picojobs.hooks.PlayerPointsHook;
008
009public class PointsImplementation extends EconomyImplementation {
010        
011        public PointsImplementation() {
012                this.requiredPlugin = Bukkit.getPluginManager().getPlugin("PlayerPoints");
013        }
014        
015        @Override
016        public String getName() {
017                return "POINTS";
018        }
019
020        @Override
021        public double getBalance(Player player) {
022                return PlayerPointsHook.getPlayerPointsAPI().look(player.getUniqueId());
023        }
024
025        @Override
026        public void deposit(Player player, double amount) {
027                PlayerPointsHook.getPlayerPointsAPI().give(player.getUniqueId(), (int)amount);
028        }
029
030        @Override
031        public void withdraw(Player player, double amount) {
032                PlayerPointsHook.getPlayerPointsAPI().take(player.getUniqueId(), (int)amount);
033        }
034
035}
