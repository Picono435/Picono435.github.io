001package com.gmail.picono435.picojobs.listeners;
002
003import java.util.HashMap;
004import java.util.Map;
005
006import org.bukkit.entity.Player;
007import org.bukkit.event.EventHandler;
008import org.bukkit.event.Listener;
009import org.bukkit.event.inventory.InventoryClickEvent;
010import org.bukkit.inventory.ItemStack;
011
012import com.gmail.picono435.picojobs.api.Job;
013import com.gmail.picono435.picojobs.api.JobPlayer;
014import com.gmail.picono435.picojobs.api.PicoJobsAPI;
015import com.gmail.picono435.picojobs.hooks.VaultHook;
016import com.gmail.picono435.picojobs.managers.LanguageManager;
017import com.gmail.picono435.picojobs.utils.FileCreator;
018
019public class ClickInventoryListener implements Listener {
020        
021        public static Map<ItemStack, String> actionItems = new HashMap<ItemStack, String>();
022        
023        @EventHandler()
024        public void onChooseJob(InventoryClickEvent e) {
025                if(e.getCurrentItem() == null || e.getCurrentItem().getItemMeta() == null || e.getCurrentItem().getItemMeta().getDisplayName() == null) return;
026                
027                /*
028                 * Choose Jobs Menu Clicking Event
029                 */
030                if(e.getView().getTitle().equals(FileCreator.getGUI().getString("gui-settings.choose-job.title"))) {
031                        e.setCancelled(true);
032                        Player p = (Player) e.getWhoClicked();
033                        JobPlayer jp = PicoJobsAPI.getPlayersManager().getJobPlayer(p);
034                        Job job = PicoJobsAPI.getJobsManager().getJobByDisplayname(e.getCurrentItem().getItemMeta().getDisplayName());
035                        if(job.requiresPermission() && !p.hasPermission("picojobs.job." + job.getName())) {
036                                p.sendMessage(LanguageManager.getMessage("no-permission", p));
037                                return;
038                        }
039                        jp.setJob(job);
040                        p.sendMessage(LanguageManager.getMessage("choosed-job", p));
041                        p.closeInventory();
042                        return;
043                }
044                
045                /*
046                 * Accept Work Menu Clicking Event
047                 */
048                if(e.getView().getTitle().equals(FileCreator.getGUI().getString("gui-settings.need-work.title"))) {
049                        e.setCancelled(true);
050                        Player p = (Player) e.getWhoClicked();
051                        JobPlayer jp = PicoJobsAPI.getPlayersManager().getJobPlayer(p);
052                        //Job job = jp.getJob();
053                        String action = actionItems.get(e.getCurrentItem());
054                        if(action.equalsIgnoreCase("salary")) {
055                                double salary = jp.getSalary();
056                                if(salary <= 0) {
057                                        p.sendMessage(LanguageManager.getMessage("no-salary", p));
058                                        p.closeInventory();
059                                        return;
060                                }
061                                if(!VaultHook.isEnabled() || !VaultHook.hasEconomyPlugin()) {
062                                        p.sendMessage(LanguageManager.formatMessage("&cWe did not find any compatible economy plugin in the server. Please contact an adminstrator, as this option does not work without it."));
063                                        p.closeInventory();
064                                        return;
065                                }
066                                p.sendMessage(LanguageManager.getMessage("got-salary", p));
067                                VaultHook.getEconomy().depositPlayer(p, salary);
068                                jp.removeSalary(salary);
069                                p.closeInventory();
070                                return;
071                        }
072                        if(action.equalsIgnoreCase("acceptwork")) {
073                                p.sendMessage(LanguageManager.getMessage("accepted-work", p));
074                                jp.setWorking(true);
075                                p.closeInventory();
076                                return;
077                        }
078                        return;
079                }
080                
081                /*
082                 * Status Work Menu Clicking Event
083                 */
084                if(e.getView().getTitle().equals(FileCreator.getGUI().getString("gui-settings.has-work.title"))) {
085                        e.setCancelled(true);
086                        Player p = (Player) e.getWhoClicked();
087                        JobPlayer jp = PicoJobsAPI.getPlayersManager().getJobPlayer(p);
088                        //Job job = jp.getJob();
089                        String action = actionItems.get(e.getCurrentItem());
090                        if(action.equalsIgnoreCase("salary")) {
091                                double salary = jp.getSalary();
092                                if(salary <= 0) {
093                                        p.sendMessage(LanguageManager.getMessage("no-salary", p));
094                                        p.closeInventory();
095                                        return;
096                                }
097                                if(!VaultHook.isEnabled() || !VaultHook.hasEconomyPlugin()) {
098                                        p.sendMessage(LanguageManager.formatMessage("&cWe did not find any compatible economy plugin in the server. Please contact an adminstrator, as this option does not work without it."));
099                                        p.closeInventory();
100                                        return;
101                                }
102                                p.sendMessage(LanguageManager.getMessage("got-salary", p));
103                                VaultHook.getEconomy().depositPlayer(p, salary);
104                                jp.removeSalary(salary);
105                                p.closeInventory();
106                                return;
107                        }
108                        return;
109                }
110        }
111}
