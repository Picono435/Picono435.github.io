001package com.gmail.picono435.picojobs.listeners;
002
003import org.bukkit.entity.Player;
004import org.bukkit.event.EventHandler;
005import org.bukkit.event.EventPriority;
006import org.bukkit.event.Listener;
007import org.bukkit.event.player.PlayerJoinEvent;
008import org.bukkit.scheduler.BukkitRunnable;
009
010import com.gmail.picono435.picojobs.PicoJobsPlugin;
011import com.gmail.picono435.picojobs.api.JobPlayer;
012import com.gmail.picono435.picojobs.managers.LanguageManager;
013
014import mkremins.fanciful.FancyMessage;
015import net.md_5.bungee.api.ChatColor;
016
017public class CreatePlayerListener implements Listener {
018        
019        @EventHandler()
020        public void onJoin(PlayerJoinEvent e) {
021                Player p = e.getPlayer();
022                if(!PicoJobsPlugin.getInstance().playersdata.containsKey(p.getUniqueId())) {
023                        PicoJobsPlugin.getInstance().playersdata.put(p.getUniqueId(), new JobPlayer(null, 0, 1, 0, false, p.getUniqueId()));
024                }
025        }
026        
027        @EventHandler(priority = EventPriority.MONITOR)
028        public void onCheckVersionJoin(PlayerJoinEvent e) {
029                Player p = e.getPlayer();
030                
031                // VERSION CHECKER
032                new BukkitRunnable() {
033                        public void run() {
034                                if(p.hasPermission("picojobs.admin") && PicoJobsPlugin.getInstance().isOldVersion()) {
035                                        String message = "\n" + LanguageManager.formatMessage("&cYou are using an old version of PicoJobs. This new version can include fixes to current errors.\n&7 You can update automatically by clicking in this message.\n&c");
036                                        new FancyMessage(message)
037                                                        .command("/jobsadmin update")
038                                                        .tooltip(ChatColor.RED + "Click here to update PicoJobs plugin to v" + PicoJobsPlugin.getInstance().getLastestPluginVersion())
039                                                        .send(p);
040                                }
041                        }
042                }.runTaskLater(PicoJobsPlugin.getInstance(), 20L);
043        }
044}
