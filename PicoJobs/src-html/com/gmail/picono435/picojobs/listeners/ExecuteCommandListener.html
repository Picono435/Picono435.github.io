001package com.gmail.picono435.picojobs.listeners;
002
003import org.bukkit.entity.Player;
004import org.bukkit.event.EventHandler;
005import org.bukkit.event.EventPriority;
006import org.bukkit.event.Listener;
007import org.bukkit.event.player.PlayerCommandPreprocessEvent;
008
009import com.gmail.picono435.picojobs.api.JobPlayer;
010import com.gmail.picono435.picojobs.api.PicoJobsAPI;
011import com.gmail.picono435.picojobs.managers.LanguageManager;
012
013public class ExecuteCommandListener implements Listener {
014        
015        @EventHandler(priority = EventPriority.LOWEST)
016        public void onCommand(PlayerCommandPreprocessEvent e) {
017                Player p = e.getPlayer();
018                String cmd = e.getMessage().split(" ")[0];
019                if(!PicoJobsAPI.getSettingsManager().getAllowedCommands().containsKey(cmd)) return;
020                int level = PicoJobsAPI.getSettingsManager().getAllowedCommands().get(cmd);
021                JobPlayer jp = PicoJobsAPI.getPlayersManager().getJobPlayer(p);
022                if(jp.getMethodLevel() < level) {
023                        p.sendMessage(LanguageManager.getMessage("need-level", p).replace("%a%", level + ""));
024                        e.setCancelled(true);
025                        return;
026                }
027        }
028}
