001package com.gmail.picono435.picojobs.listeners.jobs;
002
003import org.bukkit.entity.Player;
004import org.bukkit.event.EventHandler;
005import org.bukkit.event.Listener;
006import org.bukkit.event.player.PlayerBucketFillEvent;
007
008import com.gmail.picono435.picojobs.api.Job;
009import com.gmail.picono435.picojobs.api.JobPlayer;
010import com.gmail.picono435.picojobs.api.PicoJobsAPI;
011import com.gmail.picono435.picojobs.api.Type;
012import com.gmail.picono435.picojobs.managers.LanguageManager;
013
014public class MilkListener implements Listener {
015        
016        @EventHandler()
017        public void onTakeMilk(PlayerBucketFillEvent  e) {
018                if(e.getPlayer() == null) return;
019                if(e.getBlock().isLiquid()) return;
020                Player p = e.getPlayer();
021                JobPlayer jp = PicoJobsAPI.getPlayersManager().getJobPlayer(p);
022                if(!jp.hasJob()) return;
023                if(!jp.isWorking()) return;
024                Job job = jp.getJob();
025                if(job.getType() != Type.MILK) return;
026                if(jp.simulateEvent()) {
027                        p.sendMessage(LanguageManager.getMessage("finished-work", p));
028                }
029        }
030}
