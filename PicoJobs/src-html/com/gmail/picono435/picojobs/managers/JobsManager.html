001package com.gmail.picono435.picojobs.managers;
002
003import java.util.Collection;
004
005import com.gmail.picono435.picojobs.PicoJobsPlugin;
006import com.gmail.picono435.picojobs.api.Job;
007
008public class JobsManager {
009        
010        @SuppressWarnings("unused")
011        private PicoJobsPlugin plugin;
012        
013        /**
014         * Create the manager of jobs, with it you can change everything of jobs.
015         * 
016         * @param main - the main class of the plugin.
017         * @author Picono435
018         */
019        public JobsManager(PicoJobsPlugin main) {
020                plugin = main;
021        }
022        
023        /**
024         * Gets a job with jobname
025         * 
026         * @param jobname - the job name
027         * @return the job
028         * @author Picono435
029         */
030        public Job getJob(String jobname) {
031                if(!PicoJobsPlugin.jobs.containsKey(jobname)) {
032                        return null;
033                }
034                return PicoJobsPlugin.jobs.get(jobname);
035        }
036        
037        /**
038         * Gets a job with job displayname
039         * 
040         * @param displayname - the job display name
041         * @return the job
042         * @author Picono435
043         */
044        public Job getJobByDisplayname(String displayname) {
045                for(Job job : getJobs()) {
046                        if(job.getDisplayName().equals(displayname)) {
047                                return job;
048                        }
049                }
050                return null;
051        }
052        
053        /**
054         * Gets all the jobs
055         * 
056         * @return all the jobs
057         * @author Picono435
058         */
059        public Collection<Job> getJobs() {
060                return PicoJobsPlugin.jobs.values();
061        }
062}
