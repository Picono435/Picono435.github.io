001package com.gmail.picono435.picojobs.managers;
002
003import java.io.File;
004import java.io.IOException;
005
006import org.bukkit.ChatColor;
007import org.bukkit.configuration.InvalidConfigurationException;
008import org.bukkit.configuration.file.FileConfiguration;
009import org.bukkit.configuration.file.YamlConfiguration;
010import org.bukkit.entity.Player;
011
012import com.gmail.picono435.picojobs.PicoJobsPlugin;
013import com.gmail.picono435.picojobs.hooks.PlaceholdersHook;
014
015public class LanguageManager {
016        
017        private static FileConfiguration language;
018        private static File language_file;
019    
020        /**
021         * Get a message from the language file without using player placeholders (PlaceholderAPI)
022         * 
023         * @param message - the message key
024         * @return the formatted message value
025         * @author Picono435
026         *
027         */
028    public static String getMessage(String message) {
029        String chat = language.getString(message);
030        if(chat == null) {
031                chat = "&cThe asked message was not found in the language file. Please contact an adminstrator of the server.";
032        }
033        return getPrefix() + PlaceholdersHook.setPlaceholders(null, ChatColor.translateAlternateColorCodes('&', chat));
034    }
035    
036        /**
037         * Get a message from the language file using player placeholders (PlaceholderAPI)
038         * 
039         * @param message - the message key
040         * @param p - the player
041         * @return the formatted message value
042         * @author Picono435
043         *
044         */
045    public static String getMessage(String message, Player p) {
046        String chat = language.getString(message);
047        if(chat == null) {
048                chat = "&cThe asked message was not found in the language file. Please contact an adminstrator of the server.";
049        }
050        return getPrefix() + PlaceholdersHook.setPlaceholders(p, ChatColor.translateAlternateColorCodes('&', chat));
051    }
052    
053        /**
054         * Gets the prefix of the plugin declared on the config
055         * 
056         * @return the formatted prefix
057         * @author Picono435
058         *
059         */
060    public static String getPrefix() {
061        return ChatColor.translateAlternateColorCodes('&', PicoJobsPlugin.getPlugin().getConfig().getString("prefix"));
062    }
063    
064    /**
065         * Get a message from the language file without using player placeholders (PlaceholderAPI)
066         * 
067         * @param message - the message
068         * @return the formatted message
069         * @author Picono435
070         *
071         */
072    public static String formatMessage(String message) {
073        if(message == null) {
074                message = "&cThe asked message was not found in the language file. Please contact an adminstrator of the server.";
075        }
076        return getPrefix() + PlaceholdersHook.setPlaceholders(null, ChatColor.translateAlternateColorCodes('&', message));
077    }
078    
079    /**
080         * Gets a message format from the language config file
081         * 
082         * @param message - the message key
083         * @param p - the player
084         * @return the formatted message value
085         * @author Picono435
086         *
087         */
088    public static String getFormat(String message, Player p) {
089        String chat = language.getString(message);
090        if(chat == null) {
091                chat = "&cThe asked message was not found in the language file. Please contact an adminstrator of the server.";
092        }
093        return PlaceholdersHook.setPlaceholders(p, ChatColor.translateAlternateColorCodes('&', chat));
094    }
095    
096    /**
097         * Creates and saves thew Language Config File
098         * 
099         * @author Picono435
100         *
101         */
102    public static void createLanguageFile() {
103        String lang = PicoJobsPlugin.getPlugin().getConfig().getString("lang");
104        language_file = new File(PicoJobsPlugin.getPlugin().getDataFolder(), "langs" + File.separatorChar + lang + ".yml");
105        if (!language_file.exists()) {
106                language_file.getParentFile().mkdirs();
107                PicoJobsPlugin.getPlugin().saveResource("langs" + File.separatorChar + lang + ".yml", false);
108         }
109
110        language = new YamlConfiguration();
111        try {
112            language.load(language_file);
113        } catch (IOException | InvalidConfigurationException e) {
114            e.printStackTrace();
115        }
116    }
117}
