001package com.gmail.picono435.picojobs.managers;
002
003import java.util.UUID;
004
005import org.bukkit.Bukkit;
006import org.bukkit.entity.Player;
007
008import com.gmail.picono435.picojobs.PicoJobsPlugin;
009import com.gmail.picono435.picojobs.api.JobPlayer;
010
011public class PlayersManager {
012        
013        @SuppressWarnings("unused")
014        private static PicoJobsPlugin plugin;
015        
016        public PlayersManager(PicoJobsPlugin main) {
017                plugin = main;
018        }
019        
020        /**
021         * Get the JobPlayer object of a Player
022         * 
023         * @param p - the player
024         * @return the JobPlayer object, returns null if there is no JobPlayer
025         * @author Picono435
026         */
027        public JobPlayer getJobPlayer(Player p) {
028                return PicoJobsPlugin.playersdata.get(p.getUniqueId());
029        }
030        
031        /**
032         * Get the JobPlayer object of a Player
033         * 
034         * @param uuid - the UUID of the player
035         * @return the JobPlayer object, returns null if there is no JobPlayer
036         * @author Picono435
037         */
038        public JobPlayer getJobPlayer(UUID uuid) {
039                return PicoJobsPlugin.playersdata.get(uuid);
040        }
041        
042        /**
043         * Get the JobPlayer object of a Player
044         * 
045         * @param name - the name of the player
046         * @return the JobPlayer object, returns null if there is no JobPlayer
047         * @author Picono435
048         */
049        @Deprecated
050        public JobPlayer getJobPlayer(String name) {
051                return PicoJobsPlugin.playersdata.get(Bukkit.getOfflinePlayer(name).getUniqueId());
052        }
053}
