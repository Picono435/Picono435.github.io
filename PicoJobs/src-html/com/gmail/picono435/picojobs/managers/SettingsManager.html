001package com.gmail.picono435.picojobs.managers;
002
003import java.util.HashMap;
004import java.util.Map;
005
006import org.bukkit.configuration.ConfigurationSection;
007import org.bukkit.configuration.file.FileConfiguration;
008
009import com.gmail.picono435.picojobs.PicoJobsPlugin;
010
011public class SettingsManager {
012        
013        private PicoJobsPlugin plugin;
014        
015        private String prefix;
016        private String lang;
017        private String storageMethod;
018        private int saveInterval;
019        private int commandAction;
020        private Map<String, Integer> allowedCommands = new HashMap<String, Integer>();
021        private ConfigurationSection mysqlConfiguration;
022        private ConfigurationSection mongodbConfiguration;
023        
024        public SettingsManager(PicoJobsPlugin plugin) {
025                this.plugin = plugin;
026                
027                reloadConfigurations();
028        }
029        
030        /**
031         * Reload all the configurations
032         * 
033         * @return true if no errors, false if there is errors
034         */
035        public boolean reloadConfigurations() {
036                FileConfiguration config = this.plugin.getConfig();
037                this.prefix = config.getString("prefix");
038                this.lang = config.getString("lang");
039                this.storageMethod = config.getConfigurationSection("storage").getString("storage-method");
040                this.saveInterval = config.getInt("save-interval");
041                this.commandAction = config.getInt("jobs-action");
042                for(String cmd : config.getConfigurationSection("commands").getKeys(false)) {
043                        allowedCommands.put(cmd, config.getConfigurationSection("commands").getInt(cmd));
044                }
045                this.mysqlConfiguration = config.getConfigurationSection("storage").getConfigurationSection("mysql");
046                this.mongodbConfiguration = config.getConfigurationSection("storage").getConfigurationSection("mongodb");
047                return true;
048        }
049        
050        public String getPrefix() {
051                return prefix;
052        }
053        
054        public String getLanguage() {
055                return lang;
056        }
057        
058        public String getStorageMethod() {
059                return storageMethod;
060        }
061        
062        public int getSaveInterval() {
063                return saveInterval;
064        }
065        
066        public int getCommandAction() {
067                return commandAction;
068        }
069        
070        public Map<String, Integer> getAllowedCommands() {              
071                return allowedCommands;
072        }
073        
074        public ConfigurationSection getMySQLConfiguration() {
075                return mysqlConfiguration;
076        }
077        
078        public ConfigurationSection getMongoDBConfiguration() {
079                return mongodbConfiguration;
080        }
081
082        // NOT DEFINED YET
083        
084        public double getSalaryFrequency() {
085                return 0.3D;
086        }
087        
088        public double getMethodFrequency() {
089                return 0.3D;
090        }
091}
