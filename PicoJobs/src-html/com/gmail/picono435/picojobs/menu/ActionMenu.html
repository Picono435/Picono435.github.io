001package com.gmail.picono435.picojobs.menu;
002
003import java.util.Arrays;
004import java.util.List;
005
006import org.bukkit.entity.Player;
007
008public class ActionMenu {
009        
010        private Player p;
011        private ActionEnum action;
012        private List<Object> values;
013        
014        public ActionMenu(Player p, ActionEnum action, Object... values) {
015                this.p = p;
016                this.action = action;
017                this.values = Arrays.asList(values);
018        }
019        
020        public Player getPlayer() {
021                return p;
022        }
023        
024        public ActionEnum getAction() {
025                return action;
026        }
027        
028        public List<Object> getValues() {
029                return values;
030        }
031}
