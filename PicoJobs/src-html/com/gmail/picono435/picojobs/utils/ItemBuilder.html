001package com.gmail.picono435.picojobs.utils;
002
003import java.util.ArrayList;
004import java.util.Arrays;
005import java.util.List;
006import java.util.Map;
007
008import org.bukkit.Color;
009import org.bukkit.Material;
010import org.bukkit.enchantments.Enchantment;
011import org.bukkit.inventory.ItemFlag;
012import org.bukkit.inventory.ItemStack;
013import org.bukkit.inventory.meta.ItemMeta;
014import org.bukkit.inventory.meta.LeatherArmorMeta;
015import org.bukkit.inventory.meta.SkullMeta;
016
017public class ItemBuilder {
018   private ItemStack is;
019
020   public ItemBuilder(Material m){
021     this(m, 1);
022   }
023
024   public ItemBuilder(ItemStack is){
025     this.is=is;
026   }
027
028   public ItemBuilder(Material m, int amount){
029           if(m == null) {
030                   m = Material.STONE;
031           }
032     is= new ItemStack(m, amount);
033   }
034
035   @Deprecated
036   public ItemBuilder(Material m, int amount, byte durability){
037           if(m == null) {
038                   m = Material.STONE;
039           }
040     is = new ItemStack(m, amount, durability);
041   }
042
043   public ItemBuilder clone(){
044     return new ItemBuilder(is);
045   }
046   
047   @SuppressWarnings("deprecation")
048public ItemBuilder setDurability(short dur){
049     is.setDurability(dur);
050     return this;
051   }
052
053   public ItemBuilder setName(String name){
054     ItemMeta im = is.getItemMeta();
055     im.setDisplayName(name);
056     is.setItemMeta(im);
057     return this;
058   }
059
060   public ItemBuilder addUnsafeEnchantment(Enchantment ench, int level){
061     is.addUnsafeEnchantment(ench, level);
062     return this;
063   }
064
065   public ItemBuilder removeEnchantment(Enchantment ench){
066     is.removeEnchantment(ench);
067     return this;
068   }
069
070   @SuppressWarnings("deprecation")
071public ItemBuilder setSkullOwner(String owner){
072     try{
073       SkullMeta im = (SkullMeta)is.getItemMeta();
074       im.setOwner(owner);
075       is.setItemMeta(im);
076     }catch(ClassCastException expected){}
077     return this;
078   }
079
080   public ItemBuilder addEnchant(Enchantment ench, int level){
081     ItemMeta im = is.getItemMeta();
082     im.addEnchant(ench, level, true);
083     is.setItemMeta(im);
084     return this;
085   }
086
087   public ItemBuilder addEnchantments(Map<Enchantment, Integer> enchantments){
088     is.addEnchantments(enchantments);
089     return this;
090   }
091
092   @SuppressWarnings("deprecation")
093public ItemBuilder setInfinityDurability(){
094     is.setDurability(Short.MAX_VALUE);
095     return this;
096   }
097
098   public ItemBuilder setLore(String... lore){
099     ItemMeta im = is.getItemMeta();
100     im.setLore(Arrays.asList(lore));
101     is.setItemMeta(im);
102     return this;
103   }
104
105   public ItemBuilder setLore(List<String> lore) {
106     ItemMeta im = is.getItemMeta();
107     im.setLore(lore);
108     is.setItemMeta(im);
109     return this;
110   }
111
112   public ItemBuilder removeLoreLine(String line){
113     ItemMeta im = is.getItemMeta();
114     List<String> lore = new ArrayList<>(im.getLore());
115     if(!lore.contains(line))return this;
116     lore.remove(line);
117     im.setLore(lore);
118     is.setItemMeta(im);
119     return this;
120   }
121
122   public ItemBuilder removeLoreLine(int index){
123     ItemMeta im = is.getItemMeta();
124     List<String> lore = new ArrayList<>(im.getLore());
125     if(index<0||index>lore.size())return this;
126     lore.remove(index);
127     im.setLore(lore);
128     is.setItemMeta(im);
129     return this;
130   }
131
132   public ItemBuilder addLoreLine(String line){
133     ItemMeta im = is.getItemMeta();
134     List<String> lore = new ArrayList<>();
135     if(im.hasLore())lore = new ArrayList<>(im.getLore());
136     lore.add(line);
137     im.setLore(lore);
138     is.setItemMeta(im);
139     return this;
140   }
141
142   public ItemBuilder addLoreLine(String line, int pos){
143     ItemMeta im = is.getItemMeta();
144     List<String> lore = new ArrayList<>(im.getLore());
145     lore.set(pos, line);
146     im.setLore(lore);
147     is.setItemMeta(im);
148     return this;
149   }
150
151   public ItemBuilder setLeatherArmorColor(Color color){
152     try{
153       LeatherArmorMeta im = (LeatherArmorMeta)is.getItemMeta();
154       im.setColor(color);
155       is.setItemMeta(im);
156     }catch(ClassCastException expected){}
157     return this;
158   }
159
160   public ItemBuilder removeAttributes() {
161       final ItemMeta meta = this.is.getItemMeta();
162       meta.addItemFlags(ItemFlag.values());
163       this.is.setItemMeta(meta);
164       return this;
165   }
166
167   public ItemStack toItemStack(){
168     return is;
169   }
170}
