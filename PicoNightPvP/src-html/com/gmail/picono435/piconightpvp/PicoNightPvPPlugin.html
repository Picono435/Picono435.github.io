001package com.gmail.picono435.piconightpvp;
002
003import java.io.BufferedReader;
004import java.io.InputStreamReader;
005import java.net.HttpURLConnection;
006import java.net.URL;
007
008import org.apache.maven.artifact.versioning.DefaultArtifactVersion;
009import org.bstats.bukkit.Metrics;
010import org.bukkit.Bukkit;
011import org.bukkit.ChatColor;
012import org.bukkit.World;
013import org.bukkit.plugin.Plugin;
014import org.bukkit.plugin.java.JavaPlugin;
015import org.bukkit.scheduler.BukkitRunnable;
016import org.json.simple.JSONObject;
017import org.json.simple.parser.JSONParser;
018
019import com.gmail.picono435.piconightpvp.api.PicoNightPvPAPI;
020import com.gmail.picono435.piconightpvp.commands.NightPvPCommand;
021import com.gmail.picono435.piconightpvp.events.TimeChangedWorldEvent;
022import com.gmail.picono435.piconightpvp.listeners.PluginListeners;
023import com.gmail.picono435.piconightpvp.managers.LanguageManager;
024
025public class PicoNightPvPPlugin extends JavaPlugin {
026        
027        public static Plugin plugin;
028        
029        public void onEnable() {
030                plugin = this;
031                saveDefaultConfig();
032                sendConsoleMessage("[PicoNightPvP] Plugin created by: Picono435#2011. Thank you for use it.");
033                if(!verificarLicenca()) return;
034                sendConsoleMessage(ChatColor.AQUA + "[PicoNightPvP] Creating and configurating the language file selected...");
035                LanguageManager.createLanguageFile();
036                
037                int pluginId = 8043;
038        Metrics metrics = new Metrics(this, pluginId);
039        metrics.addCustomChart(new Metrics.SimplePie("premium_version", () -> "Free"));
040                
041                //FAZENDO TIMER PARA EVENTO
042                sendConsoleMessage(ChatColor.AQUA + "[PicoNightPvP] Finishing enabling the plugin...");
043                new BukkitRunnable() {
044                        @Override
045                        public void run() {
046                                for(String worldname : getConfig().getStringList("pvp-worlds")) {
047                                        World world = Bukkit.getWorld(worldname);
048                                        if(world == null) {
049                                                sendConsoleMessage(ChatColor.DARK_RED + "[PicoNightPvP] We didn't find any world with the name: " + worldname + ".");
050                                                return;
051                                        }
052                                        if(!isCurrentlyNight(worldname) && PicoNightPvPAPI.isNight(world)) {
053                                                PicoNightPvPAPI.setNight(false, world);
054                                                TimeChangedWorldEvent event = new TimeChangedWorldEvent(world, world.getTime(), false);
055                                                Bukkit.getScheduler().runTask(getPlugin(), () -> Bukkit.getPluginManager().callEvent(event));
056                                        }
057                                        if(isCurrentlyNight(worldname) && !PicoNightPvPAPI.isNight(world)) {
058                                                PicoNightPvPAPI.setNight(true, world);
059                                                TimeChangedWorldEvent event = new TimeChangedWorldEvent(world, world.getTime(), true);
060                                                Bukkit.getScheduler().runTask(getPlugin(), () -> Bukkit.getPluginManager().callEvent(event));
061                                        }
062                                }
063                        }
064                }.runTaskTimerAsynchronously(this, 0, 1);
065                
066                //REGISTRANDO EVENTOS
067                Bukkit.getPluginManager().registerEvents(new PluginListeners(), this);
068                this.getCommand("nightpvp").setExecutor(new NightPvPCommand());
069                sendConsoleMessage(ChatColor.GREEN + "[PicoNightPvP] The plugin was succefully enabled.");
070                
071                checkVersion();
072        }
073        
074        public void onDisable() {
075                sendConsoleMessage(ChatColor.GREEN + "[PicoNightPvP] The plugin was succefully disabled.");
076        }
077        
078        private static boolean isCurrentlyNight(String worldname) {
079                long time = PicoNightPvPPlugin.getPlugin().getServer().getWorld(worldname).getTime();
080                
081                if(time > 0 && time < 12300) {
082                        return false;
083                } else {
084                        return true;
085                }
086        }
087        
088        public static Plugin getPlugin() {
089                return plugin;
090        }
091        
092        public static void sendConsoleMessage(String message) {
093                PicoNightPvPPlugin.getPlugin().getServer().getConsoleSender().sendMessage(message);
094        }
095        
096        public static boolean isPremium() {
097                return false;
098        }
099        
100        private boolean verificarLicenca() {
101                sendConsoleMessage(ChatColor.YELLOW + "[PicoNightPvP] You are using the FREE version of the plugin!");
102                sendConsoleMessage(ChatColor.YELLOW + "[PicoNightPvP] Want to buy the premium version? Buy it in our site.");
103                sendConsoleMessage(ChatColor.YELLOW + "[PicoNightPvP] Our site is: https://piconodev.tk/plugins/premium");
104                return true;
105        }
106        
107        private void checkVersion() {
108                String version = "1.0";
109                try {
110            URL url = new URL("https://api.github.com/repos/Picono435/PicoNightPvP/releases/latest");
111            HttpURLConnection con = (HttpURLConnection) url.openConnection();
112            con.setRequestMethod("GET");
113            con.setConnectTimeout(5000);
114            con.setReadTimeout(5000);
115            con.setInstanceFollowRedirects(false);
116
117            BufferedReader in = new BufferedReader(new InputStreamReader(con.getInputStream()));
118            String inputLine;
119            StringBuffer content = new StringBuffer();
120            while ((inputLine = in.readLine()) != null) {
121                content.append(inputLine);
122            }
123            in.close();
124
125            JSONParser parser = new JSONParser();
126            JSONObject json = (JSONObject) parser.parse(content.toString());
127            
128            version = (String)json.get("tag_name");
129                        
130                } catch(Exception ex) {
131                        sendConsoleMessage(ChatColor.DARK_RED + "[PicoNightPvP] Could not get the lastest version.");
132                        return;
133                }
134                try {
135                        DefaultArtifactVersion pluginVesion = new DefaultArtifactVersion(getPlugin().getDescription().getVersion());
136                        DefaultArtifactVersion lastestVersion = new DefaultArtifactVersion(version);
137                        if(lastestVersion.compareTo(pluginVesion) > 0) {
138                                new BukkitRunnable() {
139                                        public void run() {
140                                                sendConsoleMessage(ChatColor.DARK_RED + "[PicoNightPvP] You are using a old version of the plugin. Please download the new version in our pages.");
141                                                return;
142                                        }
143                                }.runTaskLater(this, 5L);
144                        } else {
145                                new BukkitRunnable() {
146                                        public void run() {
147                                                sendConsoleMessage(ChatColor.GREEN + "[PicoNightPvP] You are using the lastest version of the plugin.");
148                                                return;
149                                        }
150                                }.runTaskLater(this, 5L);
151                        }
152                } catch (Exception e) {
153                        sendConsoleMessage(ChatColor.DARK_RED + "[PicoNightPvP] Could not get the lastest version.");
154                        return;
155                }
156        }
157}
